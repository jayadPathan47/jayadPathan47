name: Generate contribution snake

on:
  schedule:
    # Run once every 24 hours
    - cron: "0 0 * * *"
  workflow_dispatch:
  # Also run on every push to main/master branch
  push:
    branches:
      - main
      - master

jobs:
  generate:
    runs-on: ubuntu-latest
    timeout-minutes: 10
    
    steps:
      # Generates a game from a github user contributions graph, output a svg animation at <svg_out_path>
      - name: Generate github-contribution-grid-snake
        uses: Platane/snk/svg-only@v3
        with:
          github_user_name: iamgajanan70
          outputs: |
            dist/github-contribution-grid-snake.svg
            dist/github-contribution-grid-snake-dark.svg?palette=github-dark&color_snake=#ff2a85&color_dots=#0a0512,#2a133f,#5a257f,#913abf,#ff2a85
            
      # Push the content of <build_dir> to a branch
      # The content will be available at https://raw.githubusercontent.com/<username>/<repo>/<branch>/<path>
      - name: Push github-contribution-grid-snake to output branch
        uses: crazy-max/ghaction-github-pages@v3.1.0
        with:
          target_branch: output
          build_dir: dist
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
